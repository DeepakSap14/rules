/**
    Rules can work also with nested object where you can
    define how they should be rendered.

    Here it says. When you render a footer field which is
    type of XYZ class use my custom component for it
    and place it at bottom zone

                */

case=nested class=Invoice {
    field=footer {
        component:NestedMetaComponent;
        bindings:{
            header:$value;
            useNoLabelLayout:true;
        }
    }
    zNone => *;
    zLeft => uniqueName => itemPrice => supplier => requestor;
    zBottom => footer;

}
